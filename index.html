function generateResponse(input) { const q = input.toLowerCase(); // 1. FAQ Search const faq = DB.FAQ?.find(f => q.includes(f.Question?.toLowerCase()) || q.includes(f.Category?.toLowerCase())); if (faq) return `${faq.Question}
${faq.Answer}`; // 2. Agenda / Sessions (Using "Session Title" with space) const session = DB.Agenda?.find(s => q.includes(s["Session Title"]?.toLowerCase()) || q.includes(s["Speaker Name(s)"]?.toLowerCase())); if (session) return `${session["Session Title"]}
📅 ${session.Date} | ${session["Start Time"]}
📍 Room: ${session.Room}
🎤 ${session["Speaker Name(s)"]}`; // 3. Speakers (Using "Full Name" with space) const spk = DB.Speakers?.find(s => q.includes(s["Full Name"]?.toLowerCase())); if (spk) return `Speaker: ${spk["Full Name"]}
🎓 ${spk.Organization}
🎤 Topic: ${spk["Presentation Topic"]}`; // 4. General / Registration / Logistics (Matching your specific "General Info" columns) const gen = DB.General?.find(g => q.includes(g["General Info"]?.toLowerCase())); if (gen) return `${gen["General Info"]}:
${gen.Answer}`; const reg = DB.Registration?.find(r => q.includes(r["Registration Info"]?.toLowerCase())); if (reg) return `${reg["Registration Info"]}:
${reg.Answer}`; const log = DB.Logistics?.find(l => q.includes(l["Logistics Info"]?.toLowerCase()) || q.includes('wifi')); if (log) return `${log["Logistics Info"]}:
${log.Answer}`; // 5. Exhibitors const exh = DB.Exhibitors?.find(e => q.includes(e["Company Name"]?.toLowerCase())); if (exh) return `${exh["Company Name"]}
📍 Booth: ${exh["Booth Number"]}
🏢 Zone: ${exh["Hall / Zone"]}`; // 6. Rooms const room = DB.Rooms?.find(r => q.includes(r["Room / Space Name"]?.toLowerCase())); if (room) return `Room: ${room["Room / Space Name"]}
🏢 Level: ${room["Floor / Level"]}
👥 Capacity: ${room.Capacity}`; // 7. Social / Hotels / Transport const soc = DB.Social?.find(s => q.includes(s["Event Name"]?.toLowerCase()) || q.includes('social')); if (soc) return `${soc["Event Name"]}
📍 ${soc["Location / Venue"]}
📅 ${soc.Date} at ${soc["Start Time"]}`; const hotel = DB.Hotels?.find(h => q.includes(h.Name?.toLowerCase()) || q.includes('hotel')); if (hotel) return `${hotel.Name}
📍 ${hotel.Address}
🚶 ${hotel["Distance to Venue"]}`; const trans = DB.Transport?.find(t => q.includes(t["Transport General"]?.toLowerCase()) || q.includes('airport') || q.includes('taxi')); if (trans) return `${trans["Transport General"]}:
${trans.Answer}`; return "I'm not quite sure about that. Please visit the **Registration Desk** in the MTCC North Lobby or check the **Conference4me** app for live updates."; }
